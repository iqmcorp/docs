import {
  getInstance
} from "prerequisites"

const axios = getInstance();

interface InsightCategoryDetails {
  rate: number;
  enabled: boolean;
  markupType: string;
  markupTypeId: number;
  markupValue: number;
}

interface Responses {
  200: {
    headers: {
      [name: string]: unknown;
    };
    content: {
      "application/json": {
        success: boolean;
        data: {
          owId: number;
          categories: Record<string, InsightCategoryDetails>;
        }
      }
    };
  };
}

function getInsightFees(): Promise<Responses> {
  const options = {
    method: 'GET',
    url: 'https://api.iqm.com/api/v3/fa/customer/{customerOwId}/insight-fees',
    params: {
      path: {
        customerOwId: `number`
      },
      query: {
        type: `string`
      }
    }
  };

  return axios.request(options).then(({ data }: { data: Responses }) => data);
}

### Update Customer Insights Fees Details

<CopyUrl method="PATCH" url="https://api.iqm.com/api/v3/fa/customer/{customerOwId}/insight-fees" />

<div class="endpoint-container">
  <div class="child1">

Update insight fees for a specific category for a Customer.

| Path Parameters | |
| ---- | --- |
| `customerOwId` <br /><span class="type-text">integer</span> | Customer Organization Workspace ID |

| Request Schema |  |
| ---- | --- |
| `category` <br /><span class="type-text">string</span> <span class="required-text">required</span> | Insight report type category. Valid values: <var>pld</var>, <var>vld</var>, <var>ict</var>, <var>sls</var>, <var>aqs</var>, <var>nld</var> |
| `rate` <br /><span class="type-text">number</span> | Rate for the insight report |
| `enabled` <br /><span class="type-text">boolean</span> | Enable or disable the report |
| `markupType` <br /><span class="type-text">string</span> | Markup type: <var>Absolute</var> or <var>Percentage</var> |
| `markupTypeId` <br /><span class="type-text">integer</span> | Markup type ID |
| `markupValue` <br /><span class="type-text">number</span> | Markup value |

import {
  getInstance
} from "prerequisites"

const axios = getInstance();

interface Responses {
  200: {
    headers: {
      [name: string]: unknown;
    };
    content: {
      "application/json": {
        success: boolean;
        data: {
          id: number;
          message: string;
        }
      }
    };
  };
}

function updateInsightFees(): Promise<Responses> {
  const options = {
    method: 'PATCH',
    url: 'https://api.iqm.com/api/v3/fa/customer/{customerOwId}/insight-fees',
    params: {
      path: {
        customerOwId: `number`
      }
    },
    requestBody: {
      content: {
        "application/json": {
          category: `string`,
          rate?: `number`,
          enabled?: `boolean`,
          markupType?: `string`,
          markupTypeId?: `number`,
          markupValue?: `number`,
        }
      }
    }
  };

  return axios.request(options).then(({ data }: { data: Responses }) => data);
}
